req.params,req.query,req.body的区别

req.params

用在get请求当中
url： http://localhost:3000/books/book/1
使用：把请求 / 后面的参数当成id，通过req.params就能获取到id
实例：

app.get('/books/book/:id',(req,res)=>{
    let id = req.params.id;
});

req.query

用在get请求当中
url：http://localhost:3000/toEditBook?id=1
实例：

app.get("/toEditBook",function (req,res) {
    res.send(req.query["id"]);
});

req.body

用在post请求当中
需要使用body-parser包

const bodyParser = require('body-parser');
// 处理请求参数
// 挂载参数处理中间件（post）
app.use(bodyParser.urlencoded({ extended: false }));
// 处理json格式的参数，不是必须的
app.use(bodyParser.json());

一些tips

res.json()

真实项目中，服务器端大多采用JSON对象作为响应数据的格式。
res.json([body])发送JSON响应。
该方法res.send()与将对象或数组作为参数相同。
即res.json()就是 res.send([body])方法中参数是Array 或 Object的情况， ‘content-type’: ‘application/json’。

参考链接：https://blog.csdn.net/m0_37263637/article/details/79753342

href=”javascript:;

“href=”javascript:;”表示去掉a标签的默认行为。
“javascript:”是表示在触发<a>默认动作时，执行一段JavaScript代码，而 javascript:; 表示什么都不执行，这样点击<a>时就没有任何反应。一般在这种情况下，会给绑定一个事件回调，来执行业务。

项目部分js代码解析

$(function(){
    // 初始化数据列表
    function initList(){
        $.ajax({
            type : 'get',
            url : '/books',
            // 回调函数得到数据类型
            dataType : 'json',
            //  success(result,status,xhr)    当请求成功时运行的函数。
            // 第一个参数result是请求返回的结果，是一个json数组
            success : function(data){
                // 渲染数据列表，得到html内容
                // data数组里面的每个值是一个对象，list是data的别名
                var html = template('indexTpl',{list : data});
                // .html方法会识别html标签，.text方法会那内容直接当成字符串，并不会识别html标签。
                $('#dataList').html(html);
                // 必须在渲染完成内容之后才可以操作DOM标签

                $('#dataList').find('tr').each(function(index,element){
                    // :eq（index）  
                    // 获取到的td元素中，选择索引号为5的元素，索引号index从0开始。
                    var td = $(element).find('td:eq(5)'); // 获取编辑/删除按钮
                    var id = $(element).find('td:eq(0)').text(); //获取id
                    // 绑定编辑图书的单击事件
                    td.find('a:eq(0)').click(function(){
                        editBook(id);
                    });
                    // 绑定删除图书的单击事件
                    td.find('a:eq(1)').click(function(){
                        deleteBook(id);
                    });

                    // 绑定添加图书信息的单击事件
                    addBook();
                    // 重置表单
                    var form = $('#addBookForm'); //iquery对象
                    form.get(0).reset(); //转为dom对象并调用原生方法
                    // type为hidden的元素无法用reset()重置
                    form.find('input[type=hidden]').val(''); 
                });
            }
        });
    }
    initList();
    // 删除图书信息
    function deleteBook(id){
        $.ajax({
            type : 'delete',
            url : '/books/book/' + id,
            dataType : 'json',
            success : function(data){
                // 删除图书信息之后重新渲染数据列表
                if(data.flag == '1'){
                    initList();
                }
            }
        });
    }

    // 编辑图书信息
    function editBook(id){
        var form = $('#addBookForm');
        // 先根据数据id查询最新的数据
        $.ajax({
            type : 'get',
            url : '/books/book/' + id,
            dataType : 'json',
            success : function(data){
                // 初始化弹窗
                var mark = new MarkBox(600,400,'编辑图书',form.get(0));
                mark.init();
                // 填充表单数据
                form.find('input[name=id]').val(data.id); //val()是赋值操作
                form.find('input[name=name]').val(data.name);
                form.find('input[name=author]').val(data.author);
                form.find('input[name=category]').val(data.category);
                form.find('input[name=description]').val(data.description);
                // 对表单提交按钮重新绑定单击事件
                form.find('input[type=button]').unbind('click').click(function(){
                    // 编辑完成数据之后重新提交表单
                    $.ajax({
                        type : 'put',
                        url : '/books/book',
                        // .serialize() 方法创建以标准 URL 编码表示的文本字符串。它的操作对象是代表表单元素集合的 jQuery 对象。
                        data : form.serialize(), 
                        dataType : 'json',
                        success : function(data){
                            if(data.flag == '1'){
                                // 隐藏弹窗
                                mark.close();
                                // 重新渲染数据列表
                                initList();
                            }
                        }
                    });
                });
            }
        });
    }

    // 添加图书信息
    function addBook(){
        $('#addBookId').click(function(){
            var form = $('#addBookForm');
            // 实例化弹窗对象
            var mark = new MarkBox(600,400,'添加图书',form.get(0));
            mark.init();
            form.find('input[type=button]').unbind('click').click(function(){
                $.ajax({
                    type : 'post',
                    url : '/books/book',
                    data : form.serialize(),
                    dataType : 'json',
                    success : function(data){
                        if(data.flag == '1'){
                            // 关闭弹窗
                            mark.close();
                            // 添加图书成功之后重新渲染数据列表
                            initList();
                        }
                    }
                });
            });
        });
    }

    // 查询天气
    $('#weather').click(function(){
        var cityCode = $('select option:selected').val();
        $.ajax({
            type : 'get',
            url : '/weather/' + cityCode,
            dataType : 'json',
            success : function(data){
                var html = template('weatherTpl',data.info);
                var mark = new MarkBox(600,400,'天气信息',$(html).get(0));
                mark.init();
            }
        });
    });


});
